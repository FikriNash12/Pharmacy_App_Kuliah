-- Membuat Table Obat --
CREATE TABLE obat (
    id SERIAL PRIMARY KEY,
    nama VARCHAR(100),
    stok INT,
    harga NUMERIC(10,2)
);

-- Menambahkan Kolom Baru --
alter table obat
add column kategori varchar(50);

alter table obat
add column tanggal_kadaluarsa date;

-- Melihat Table --
select * from obat;

-- Membuat table user Untuk Login --
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL
);

-- Melihat tabel user --
select * from users

-- Membuat table riwayat
CREATE TABLE riwayat (
    id SERIAL PRIMARY KEY,
    username VARCHAR(100),
    aksi VARCHAR(50),      -- Contoh: "Tambah", "Edit", "Hapus"
    deskripsi TEXT,        -- Contoh: "Menambah obat Paracetamol"
    waktu TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Melihat table riwayat
select * from riwayat;
